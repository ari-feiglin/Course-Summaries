    \_xp\def\csname _b#1\endcsname[##1]{

            title={%

                default=\novalue%
            },
            name={%
                name=_name,

            }%
        }{##1}%
        \bppbox{#3}{#4}{#5}%
        \ifdomcount{\globalsetters\advancecounter{math counter}{1}\localsetters}\fi%
        \def\curremphcolor{#5}%
        \hbox to\hsize{%
            \emphcolor\ifdomcount\printmcount{} \fi#2%

                \ (\_title)%
            \fi%
            \ifdomcount%

                    \_xp\anchor\_xp{\_name}%
                    \_xp\xdef\csname math@\_name\endcsname{\printmcount}%
                    \unless\ifx\_title\novalue%
                        \_xp\xdef\csname math@\_name\endcsname{\_title}%
                        \addthreedepth{\_title}{\_name}%
                    \fi%
                \fi%
            \fi%
        \hfil}%
        \hrule height\z@\relax%
        \medskip%
    }%
    \_xp\edef\csname b#1\endcsname{\medskip\noexpand\_ifnextchar[ {\_xp\noexpand\csname _b#1\endcsname}{\_xp\noexpand\csname _b#1\endcsname[]}}%

\def\@mathify@col#1#2{%
    \relax\xdef#1{%
        #1%
        \ifx\@mathifyfstcol\@false%
            &%
        \fi%
        $#2$%
    }%
    \ifx\@mathifyfstcol\@true%
        \global\let\@mathifyfstcol\@false%
    \fi%
}

\def\@mathify@row#1{%
    \noalign{\xdef#1{%
        #1%
        \ifx\@mathifyfstrow\@false%
            \noexpand\cr%
        \fi%
    }%
    \ifx\@mathifyfstrow\@true%
        \global\let\@mathifyfstrow\@false%
    \fi%
    \global\let\@mathifyfstcol\@true}%
}

\def\mathify#1#2{%
    \def#1{}%
    \let\@mathifyfstrow\@true%
    \let\@mathifyfstcol\@true%
    \vbox{\everycr={\@mathify@row{#1}}\halign{&\@mathify@col{#1}{##}\cr
        #2\cr
    }}%
}

\def\commsq#1#2#3#4#5{%
    \mathify\@csq{#1}%
    \centerline{\expandafter\drawdiagram\expanded{{\@csq}}{%
        \diagarrow{from={1,1}, to={1,2}, text={$#2$}, y distance=-.25cm}%
        \diagarrow{from={2,1}, to={2,2}, text={$#3$}, y distance=.25cm}%
        \diagarrow{from={1,1}, to={2,1}, text={$#4$}, x distance=-.25cm}%
        \diagarrow{from={1,2}, to={2,2}, text={$#5$}, x distance=-.25cm}%
    }}%
}
